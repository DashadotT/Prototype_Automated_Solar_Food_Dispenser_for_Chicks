void connectWiFi() {
  if (WiFi.status() == WL_CONNECTED) return;

  Serial.printf("\n📶 Connecting to %s", WIFI_SSID);
  WiFi.mode(WIFI_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);

  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 20) {
    delay(500);
    Serial.print(".");
    attempts++;
  }

  if (WiFi.status() == WL_CONNECTED) {
    wifiConnected = true;
    Serial.printf("\n✅ WiFi connected | IP: %s\n", WiFi.localIP().toString().c_str());
    configTime(GMT_OFFSET, DAYLIGHT_OFFSET, NTP_SERVER);
    logEvent("system", "WiFi connected", WiFi.localIP().toString());
  } else {
    wifiConnected = false;
    Serial.println("\n❌ WiFi failed — will retry");
  }
}

// 📏 ULTRASONIC SENSOR
// ============================================================
float measureDistance() {
  // Trigger pulse
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  long duration = pulseIn(ECHO_PIN, HIGH, 30000);
  if (duration == 0) return -1.0;

  return duration * 0.034f / 2.0f;
}

// Smoothed reading using median filter
float measureDistanceSmoothed() {
  // Take 5 readings
  float readings[5];
  for (int i = 0; i < 5; i++) {
    readings[i] = measureDistance();
    delay(10);
  }

  // Sort (bubble)
  for (int i = 0; i < 4; i++) {
    for (int j = i + 1; j < 5; j++) {
      if (readings[j] < readings[i]) {
        float tmp = readings[i];
        readings[i] = readings[j];
        readings[j] = tmp;
      }
    }
  }

  // Return median
  float median = readings[2];
  if (median < 0 || median > 400) return -1.0;
  return median;
}

String getFoodLevel(float cm) {
  if (cm < 0) return "Unknown";
  if (cm <= 5) return "Full";
  if (cm <= 10) return "Medium";
  if (cm <= 20) return "Low";
  return "Empty";
}

int getFoodPercentage(float cm) {
  float maxCm = 30.0;
  int pct = (int)(((maxCm - cm) / maxCm) * 100.0f);
  return constrain(pct, 0, 100);
}

// ============================================================
// 📤 SEND SENSOR DATA
// ============================================================
void sendSensorData(float distance, String foodLevel, int foodPct) {
  StaticJsonDocument<256> doc;
  doc["distance_cm"] = round(distance * 10) / 10.0;
  doc["food_level"] = foodLevel;
  doc["food_percentage"] = foodPct;

  String body;
  serializeJson(doc, body);

  int code = httpPOST(buildUrl("sensor_readings", ""), body);
  if (code == 201 || code == 200) {
    Serial.printf("✅ Sensor sent: %.1fcm | %s | %d%%\n", distance, foodLevel.c_str(), foodPct);
    failedRequests = 0;
  } else {
    Serial.printf("❌ Sensor send failed: %d\n", code);
    failedRequests++;
  }
}

// ============================================================
// 🔄 FETCH SERVO STATE
// ============================================================
bool fetchServoState() {
  String response = httpGET(buildUrl("servo_state", "?select=state,angle&limit=1"));
  if (response.isEmpty()) return false;

  StaticJsonDocument<128> doc;
  if (deserializeJson(doc, response) != DeserializationError::Ok) return false;
  if (!doc[0]["state"].is<const char*>()) return false;

  String newState = doc[0]["state"].as<String>();
  int angle = doc[0]["angle"] | -1;

  if (newState != currentServoState) {
    currentServoState = newState;
    if (newState == "Open" && !isDispensing) {
      // Manual open from dashboard
      openGate(angle > 0 ? angle : SERVO_OPEN_ANGLE);
    } else if (newState == "Closed") {
      closeGate(angle > 0 ? angle : SERVO_CLOSED_ANGLE);
    }
    Serial.printf("🔄 Servo → %s\n", newState.c_str());
    return true;
  }
  return false;
}

void dispenseFood(int duration, int openAngle, int closeAngle, String label) {
  if (isDispensing) {
    Serial.println("⚠️ Already dispensing");
    return;
  }

  isDispensing = true;
  Serial.printf("🍽️ Dispensing: %s (%ds)\n", label.c_str(), duration);

  // Open servo
  openGate(openAngle);
  currentServoState = "Open";
  updateServoState("Open", openAngle);

  // Log feeding start
  logFeedingEvent("automatic", openAngle, label);

  // Wait (blocking OK here, gate must stay open)
  delay(duration * 1000);

  // Close servo
  closeGate(closeAngle);
  currentServoState = "Closed";
  updateServoState("Closed", closeAngle);

  isDispensing = false;
  Serial.println("✅ Dispense complete");
}

// ============================================================
// 📅 FETCH SCHEDULES
// ============================================================
void fetchSchedules() {
  String response = httpGET(buildUrl("feeding_schedules", "?select=id,feeding_time,enabled,duration_seconds&order=feeding_time"));
  if (response.isEmpty()) return;

  StaticJsonDocument<1024> doc;
  if (deserializeJson(doc, response) != DeserializationError::Ok) return;

  scheduleCount = 0;
  for (JsonObject s : doc.as<JsonArray>()) {
    if (scheduleCount >= MAX_SCHEDULES) break;
    schedules[scheduleCount].id = s["id"] | 0;
    schedules[scheduleCount].feeding_time = s["feeding_time"].as<String>().substring(0, 5);
    schedules[scheduleCount].enabled = s["enabled"] | false;
    schedules[scheduleCount].duration_seconds = s["duration_seconds"] | DISPENSE_DURATION;
    schedules[scheduleCount].triggered_today = false;
    scheduleCount++;
  }
  Serial.printf("📅 Loaded %d schedules\n", scheduleCount);
}

// ============================================================
// ✅ PROCESS SCHEDULES
// ============================================================
void processSchedules() {
  struct tm timeInfo;
  if (!getLocalTime(&timeInfo)) return;

  char currentTime[6];
  snprintf(currentTime, sizeof(currentTime), "%02d:%02d", timeInfo.tm_hour, timeInfo.tm_min);

  // Reset triggers at midnight
  if (timeInfo.tm_hour == 0 && timeInfo.tm_min == 0 && timeInfo.tm_sec < 30) {
    for (int i = 0; i < scheduleCount; i++) {
      schedules[i].triggered_today = false;
    }
  }

  for (int i = 0; i < scheduleCount; i++) {
    if (!schedules[i].enabled) continue;
    if (schedules[i].triggered_today) continue;
    if (schedules[i].feeding_time == String(currentTime)) {
      schedules[i].triggered_today = true;
      Serial.printf("⏰ Schedule triggered: %s\n", schedules[i].feeding_time.c_str());
      dispenseFood(schedules[i].duration_seconds, SERVO_OPEN_ANGLE, SERVO_CLOSED_ANGLE,
                   "Schedule " + schedules[i].feeding_time);
    }
  }
}

// ❤️ HEARTBEAT — System Status
// ============================================================
void updateSystemStatus() {
  unsigned long uptime = (millis() - startupTime) / 1000;
  String ip = WiFi.status() == WL_CONNECTED ? WiFi.localIP().toString() : "disconnected";

  StaticJsonDocument<256> doc;
  doc["id"] = 1;
  doc["esp32_online"] = true;
  doc["wifi_status"] = ip;
  doc["uptime_seconds"] = (int)uptime;
  doc["last_seen"] = getISOTime();

  String body;
  serializeJson(doc, body);

  HTTPClient http;
  http.begin(buildUrl("system_status", "?id=eq.1"));
  addSupabaseHeaders(http);
  int code = http.PATCH(body);

  if (code == 404 || code == 0) {
    // Row doesn't exist, insert it
    http.end();
    http.begin(buildUrl("system_status", ""));
    addSupabaseHeaders(http);
    http.POST(body);
  }

  http.end();
  Serial.printf("❤️ Heartbeat | Uptime: %lus | WiFi: %s\n", uptime, ip.c_str());
}

// ============================================================
// ⚙️ FETCH SETTINGS
// ============================================================
void fetchSettings() {
  String response = httpGET(buildUrl("system_settings", "?limit=1"));
  if (response.isEmpty()) return;

  StaticJsonDocument<256> doc;
  if (deserializeJson(doc, response) != DeserializationError::Ok) return;
  if (!doc[0].is<JsonObject>()) return;

  JsonObject s = doc[0];
  SERVO_OPEN_ANGLE = s["servo_open_angle"] | SERVO_OPEN_ANGLE;
  SERVO_CLOSED_ANGLE = s["servo_closed_angle"] | SERVO_CLOSED_ANGLE;
  DISPENSE_DURATION = s["dispense_duration"] | DISPENSE_DURATION;
  REFILL_THRESHOLD = s["refill_threshold_cm"] | REFILL_THRESHOLD;
  SENSOR_INTERVAL = (s["sensor_interval"] | 5) * 1000;

  Serial.printf("⚙️ Settings: Open=%d° Closed=%d° Duration=%ds Threshold=%.1fcm\n",
                SERVO_OPEN_ANGLE, SERVO_CLOSED_ANGLE, DISPENSE_DURATION, REFILL_THRESHOLD);
}

// ============================================================
// 🕐 ISO TIME HELPER
// ============================================================
String getISOTime() {
  struct tm timeInfo;
  if (!getLocalTime(&timeInfo)) return "1970-01-01T00:00:00Z";
  char buf[30];
  strftime(buf, sizeof(buf), "%Y-%m-%dT%H:%M:%SZ", &timeInfo);
  return String(buf);
}

// ✅ SETUP

void setup() {
  Serial.begin(115200);
  delay(500);

  Serial.println("\n");
  Serial.println("╔══════════════════════════════════════╗");
  Serial.println("║  🌱 SolarFeed IoT v2.0               ║");
  Serial.println("║  Solar-Powered Smart Food Dispenser  ║");
  Serial.println("╚══════════════════════════════════════╝");

  // Pin setup
  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);
  digitalWrite(TRIG_PIN, LOW);

  // Servo init
  foodGateServo.attach(SERVO_PIN, 500, 2400);
  foodGateServo.write(SERVO_CLOSED_ANGLE);
  Serial.printf("🔒 Servo initialized: %d° (closed)\n", SERVO_CLOSED_ANGLE);

  startupTime = millis();

  // WiFi
  connectWiFi();
  delay(1000);

  if (WiFi.status() == WL_CONNECTED) {
    // Fetch settings first
    fetchSettings();
    delay(500);

    // Fetch schedules
    fetchSchedules();
    delay(500);

    // Initial sensor read
    float d = measureDistanceSmoothed();
    if (d > 0 && d < 400) {
      lastDistance = d;
      sendSensorData(d, getFoodLevel(d), getFoodPercentage(d));
    }

    // Initial status update
    updateSystemStatus();
    delay(500);

    // Get current servo state
    fetchServoState();

    logEvent("system", "ESP32 started", "v2.0 boot complete");
  }

  Serial.println("✅ Setup complete — entering main loop");
  Serial.printf("   Sensor every: %dms\n", SENSOR_INTERVAL);
  Serial.printf("   Servo check:  %dms\n", (int)SERVO_CHECK_INTERVAL);
  Serial.printf("   Heartbeat:    %dms\n", (int)HEARTBEAT_INTERVAL);
}

// ============================================================
// 🔄 MAIN LOOP
// ============================================================
void loop() {
  unsigned long now = millis();

  // ── WiFi reconnect ──────────────────────────────────────
  if (WiFi.status() != WL_CONNECTED) {
    if (now - wifiReconnectTime >= WIFI_RECONNECT_INTERVAL) {
      wifiReconnectTime = now;
      wifiConnected = false;
      Serial.println("📶 WiFi lost — reconnecting...");
      connectWiFi();
    }
    delay(100);
    return;  // Skip everything else until WiFi is back
  }

  // ── Sensor reading ──────────────────────────────────────
  if (now - lastSensorRead >= (unsigned long)SENSOR_INTERVAL) {
    lastSensorRead = now;

    float distance = measureDistanceSmoothed();
    if (distance > 0 && distance < 400) {
      lastDistance = distance;
      String level = getFoodLevel(distance);
      int pct = getFoodPercentage(distance);

      sendSensorData(distance, level, pct);

      // Low food alert
      if (distance > REFILL_THRESHOLD) {
        Serial.printf("⚠️ LOW FOOD! %.1fcm > %.1fcm threshold\n", distance, REFILL_THRESHOLD);
      }

      Serial.printf("📏 Distance: %.1fcm | %s | %d%%\n", distance, level.c_str(), pct);
    } else {
      Serial.println("⚠️ Invalid sensor reading");
    }
  }

  // ── Servo state check (fast — every 1s) ─────────────────
  if (!isDispensing && now - lastServoCheck >= SERVO_CHECK_INTERVAL) {
    lastServoCheck = now;
    fetchServoState();
  }

  // ── Schedule processing ──────────────────────────────────
  if (now - lastScheduleCheck >= 10000) {  // every 10s
    lastScheduleCheck = now;
    processSchedules();
  }

  // ── Refresh schedules from Supabase ─────────────────────
  if (now - lastScheduleFetch >= SCHEDULE_CHECK_INTERVAL) {
    lastScheduleFetch = now;
    fetchSchedules();
  }

  // ── Settings refresh ─────────────────────────────────────
  if (now - lastSettingsFetch >= SETTINGS_FETCH_INTERVAL) {
    lastSettingsFetch = now;
    fetchSettings();
  }

  // ── Heartbeat ────────────────────────────────────────────
  if (now - lastHeartbeat >= HEARTBEAT_INTERVAL) {
    lastHeartbeat = now;
    updateSystemStatus();
  }

  delay(50);
}
